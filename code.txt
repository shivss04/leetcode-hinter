class Solution:
    def minReorder(self, n: int, connections: List[List[int]]) -> int:
        
        def dfs(graph, start, visited = None):
            if visited is None:
                visited = set()
            visited.add(start)

            
            for next_node in graph[start] - visited:
                dfs(graph, next_node, visited)
            
            return visited
        
        dfs(connections, 0)
